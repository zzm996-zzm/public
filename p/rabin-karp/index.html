func longestDupSubstring(s string) string {

    var p,h []int

        p[i] = p[i-1] * power
    }

    substrHash := func(left,right int,h,p []int) int{
        return h[right]-h[left] * p[right-left]

        var n int = len(s)
        m := make(map[int]int)
        for i:=0;i+tl < n ;i++{
            right := tl + i + 1
            hash := substrHash(i,right,h,p)

                return s[i:right]

            m[hash]++
        }
